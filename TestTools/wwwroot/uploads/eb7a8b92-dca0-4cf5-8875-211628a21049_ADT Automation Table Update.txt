WpasAdtSteps.cs update to automate Admission Date and Time outside of the table lookup and add new step definition:


        [When("I complete the Admission Form with the following data")]
        public void WhenICompleteTheAdmissionFormWithTheFollowingData(Table table)
        {
            driver.WaitForReady();
            Adt.EnterAdmissionDate(Parameters.Dates.Present);
            driver.WaitForReady();
            Adt.EnterAdmissionTime(AdtParams.GetTime.GetRandomTime());
            driver.WaitForReady();

            var tableData = table.CreateInstance<AdtModel>();

            // Map table data to AdtModel properties
            tableData.Hospital = tableData.Hospital.Equals("Glangwili General Hospital") ? AdtParams.Hospitals.Hdd1 : tableData.Hospital;
            tableData.Ward = tableData.Ward.Equals("GGH - Tysul Ward") ? AdtParams.Wards.Ggh1 : tableData.Ward;
            tableData.Service = tableData.Service.Equals("Cardiology") ? AdtParams.Specialty.Spec1 : tableData.Service;
            tableData.Clinician = tableData.Clinician.Equals("Wilson, D G, Dr. (GMC:3311419)") ? AdtParams.Clinicians.CardioClin1 : tableData.Clinician;
            tableData.AdmissionMethod = tableData.AdmissionMethod.Equals("Emergency - GP") ? AdtParams.AdmissionMethod.Method5 : tableData.AdmissionMethod;
            tableData.Source = tableData.Source.Equals("Referral (GP)") ? AdtParams.AdmissionSource.Source1 : tableData.Source;

            // Pass the mapped data to the FullAdmission method
            Adt.FullAdmission(
                tableData.Hospital,
                tableData.Ward,
                tableData.Service,
                tableData.Clinician,
                tableData.AdmissionMethod,
                tableData.Source
            );
        }

        public void FullAdmission(string hospital, string ward, string service, string clinician, string method, string source)
        {
            _driver.WaitForReady();
            SelectAdmissionHospital(hospital);
            _driver.WaitForReady();
            SelectAdmissionWard(ward);
            _driver.WaitForReady();
            SelectAdmissionService(service);
            _driver.WaitForReady();
            SelectAdmissionClinician(clinician);
            _driver.WaitForReady();
            SelectAdmissionMethod(method);
            _driver.WaitForReady();
            SelectAdmissionSource(source);

        }
